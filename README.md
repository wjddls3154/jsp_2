# jsp_2 : 변수

// 정수형 변수
      var 변수하나 = 100;
      변수하나 = 150;
      document.write(변수하나 + 변수하나);
      document.write(typeof 변수하나);
      document.write("<br>");
// 문자형 변수
      var 변수둘 = "200";
      변수둘 = "250";
      document.write(변수둘 + 변수둘);
      document.write(typeof 변수둘);
      document.write("<br>");
      document.write(변수둘[0]); // 변수둘의 값 200중, 첫번째 인덱스 2 를 출력
